873 uint8_t HaptiCapMagSettings::calculateChecksum(uint8_t settings_version) {
889  uint8_t crc = 0x00; // Initial value
890 
891  // Ordered by place in memory, starting with number of motors.
892  crc = update_CRC8_byte(crc, nMotors);
893  crc = update_CRC8_float(crc, declination);
894  crc = update_CRC8_float(crc, inclination);
895  crc = update_CRC8_float(crc, sampleRate);
896  crc = update_CRC8_uint16_t(crc, pulseWidth);
897  crc = update_CRC8_float(crc, phaseOffset);
898  crc = update_CRC8_byte(crc, useCalibration);
899  crc = update_CRC8_byte(crc, gain);
900  crc = update_CRC8_byte(crc, averageRate);
901 
902  // Update the array settings - these are done in order, even though doing it in parallel would
903  // only require a single loop.
904  for (int i = 0; i < HC_MAX_NMOTORS; i++) {
905  crc = update_CRC8_float(crc, pinLocs[i]);
906  }
907 
908  for (int i = 0; i < HC_MAX_NMOTORS; i++) {
909  crc = update_CRC8_float(crc, motorCals[i]);
910  }
911 
912  return crc;
913 }

915 uint8_t update_CRC8_byte(uint8_t crc, uint8_t b) {
932  uint8_t data;
933  data = crc ^ b;
934  for (int i = 0; i < 8; i++ ) {
935  if (( data & 0x80 ) != 0 ) {
936  data <<= 1;
937  data ^= 0x07;
938  } else {
939  data <<= 1;
940  }
941  }
942  return data;
943 }
